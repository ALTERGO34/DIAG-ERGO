        // Gestion automatique des radio buttons Oui/Non mutuellement exclusifs

        document.addEventListener('DOMContentLoaded', function() {

            // Initialiser la date du jour

            const today = new Date();

            const dateStr = today.toLocaleDateString('fr-FR');

            document.getElementById('date_rapport').textContent = dateStr;

           

            // Copier automatiquement le nom de l'ergothérapeute

            document.getElementById('ergotherapeute').addEventListener('input', function() {

                document.getElementById('nom_ergo_rapport').textContent = this.value;

            });

           

            // Gestion des boutons radio mutuellement exclusifs

            const radioGroups = document.querySelectorAll('.radio-group');

            radioGroups.forEach(group => {

                const radios = group.querySelectorAll('input[type="radio"]');

                radios.forEach(radio => {

                    radio.addEventListener('change', function() {

                        if (this.checked) {

                            radios.forEach(r => {

                                if (r !== this) r.checked = false;

                            });

                        }

                    });

                });

            });

           

            // Animation au scroll

            const sections = document.querySelectorAll('.section');

            const observer = new IntersectionObserver((entries) => {

                entries.forEach(entry => {

                    if (entry.isIntersecting) {

                        entry.target.style.opacity = '1';

                        entry.target.style.transform = 'translateY(0)';

                    }

                });

            }, { threshold: 0.1 });

           

            sections.forEach(section => {

                section.style.opacity = '0';

                section.style.transform = 'translateY(20px)';

                section.style.transition = 'all 0.6s ease';

                observer.observe(section);

            });

        });

       

        // Fonction de sauvegarde

        function saveForm() {

            const loading = document.getElementById('loading');

            const successMsg = document.getElementById('successMessage');

           

            loading.style.display = 'block';

           

            const formData = {};

           

            // Sauvegarder tous les inputs

            document.querySelectorAll('input, textarea, select').forEach(element => {

                if (element.type === 'checkbox' || element.type === 'radio') {

                    formData[element.name || element.id] = element.checked;

                } else if (element.type !== 'file') {

                    formData[element.name || element.id] = element.value;

                }

            });

           

            // Sauvegarder dans localStorage

            localStorage.setItem('diagnosticFormData', JSON.stringify(formData));

           

            setTimeout(() => {

                loading.style.display = 'none';

                successMsg.style.display = 'block';

                successMsg.textContent = '✅ Formulaire sauvegardé avec succès !';

               

                setTimeout(() => {

                    successMsg.style.display = 'none';

                }, 3000);

            }, 1000);

        }

       

        // Fonction de restauration des données

        function restoreForm() {

            const savedData = localStorage.getItem('diagnosticFormData');

            if (savedData) {

                const formData = JSON.parse(savedData);

               

                Object.keys(formData).forEach(key => {

                    const element = document.querySelector(`[name="${key}"], #${key}`);

                    if (element) {

                        if (element.type === 'checkbox' || element.type === 'radio') {

                            element.checked = formData[key];

                        } else {

                            element.value = formData[key];

                        }

                    }

                });

               

                // Mettre à jour le nom de l'ergothérapeute dans le footer

                const ergoName = document.getElementById('ergotherapeute').value;

                if (ergoName) {

                    document.getElementById('nom_ergo_rapport').textContent = ergoName;

                }

            }

        }

       

        // Restaurer les données au chargement si elles existent

        document.addEventListener('DOMContentLoaded', function() {

            if (localStorage.getItem('diagnosticFormData')) {

                if (confirm('Des données sauvegardées ont été trouvées. Voulez-vous les restaurer ?')) {

                    restoreForm();

                }

            }

        });

       

        // Fonction d'envoi par email

        function sendByEmail() {

            const loading = document.getElementById('loading');

            const successMsg = document.getElementById('successMessage');

           

            loading.style.display = 'block';

           

            // Collecter toutes les données du formulaire

            let emailContent = 'DIAGNOSTIC AMÉNAGEMENT DU DOMICILE\n\n';

            emailContent += '=================================\n\n';

           

            // Parcourir toutes les sections

            document.querySelectorAll('.section').forEach(section => {

                const sectionTitle = section.querySelector('.section-title').textContent;

                emailContent += `\n${sectionTitle}\n${'='.repeat(sectionTitle.length)}\n\n`;

               

                // Collecter les valeurs des inputs dans cette section

                section.querySelectorAll('input, textarea, select').forEach(element => {

                    if (element.type === 'checkbox' || element.type === 'radio') {

                        if (element.checked) {

                            const label = element.parentElement.textContent.trim();

                            emailContent += `- ${label}\n`;

                        }

                    } else if (element.value && element.type !== 'file') {

                        const label = element.previousElementSibling?.textContent || element.placeholder || element.name;

                        emailContent += `${label}: ${element.value}\n`;

                    }

                });

            });

           

            // Encoder le contenu pour l'URL mailto

            const subject = encodeURIComponent('Diagnostic Aménagement du Domicile - ' + document.getElementById('beneficiaire').value);

            const body = encodeURIComponent(emailContent);

           

            setTimeout(() => {

                loading.style.display = 'none';

               

                // Ouvrir le client mail

                window.location.href = `mailto:contact@altergo34.fr?subject=${subject}&body=${body}`;

               

                successMsg.style.display = 'block';

                successMsg.textContent = '✅ Préparation de l\'email terminée !';

               

                setTimeout(() => {

                    successMsg.style.display = 'none';

                }, 3000);

            }, 1000);

        }

        // Validation des champs obligatoires

        function validateForm() {

            const requiredFields = ['beneficiaire', 'ergotherapeute', 'email_ergo', 'tel_ergo'];

            let isValid = true;

           

            requiredFields.forEach(fieldId => {

                const field = document.getElementById(fieldId);

                if (!field.value.trim()) {

                    field.style.borderColor = '#e74c3c';

                    isValid = false;

                } else {

                    field.style.borderColor = '#ddd';

                }

            });

           

            if (!isValid) {

                alert('Veuillez remplir tous les champs obligatoires (marqués en rouge)');

            }

           

            return isValid;

        }

       

        // Ajouter la validation avant la sauvegarde et l'envoi

        const originalSaveForm = window.saveForm;

        window.saveForm = function() {

            if (validateForm()) {

                originalSaveForm();

            }

        };

       

        const originalSendByEmail = window.sendByEmail;

        window.sendByEmail = function() {

            if (validateForm()) {

                originalSendByEmail();

            }
